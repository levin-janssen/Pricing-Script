SELECT kfa.wert2 AS land, t1.kundennummer, t1.produktid
FROM kunden_felder_werte kfw
JOIN (
    SELECT bestellungsid, kundennummer, einzelpreis, produktid
    FROM bestellungen_positionen 
    WHERE produktid = (
        SELECT produktid 
        FROM produkte_felder_werte 
        WHERE feldid = 44 AND wert1 = '8-21-2-1141'
    ) 
    ORDER BY datum DESC 
    LIMIT 3
) t1 ON t1.kundennummer = kfw.kundennummer
JOIN kunden_felder_auswahl kfa ON kfa.auswahlid = kfw.wert1
WHERE kfa.feldid = 48
AND kfw.feldid = 48
